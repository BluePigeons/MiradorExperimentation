      pathData += 'M' + initialPoint.x + ',' + initialPoint.y;
      pathData += ' Q' + initialPoint.x + ',' + (initialPoint.y - size);
      pathData += ' ' + (initialPoint.x + size) + ',' + (initialPoint.y - 2 * size);
      pathData += ' A' + size + ',' + size + ' 0 1 0';
      pathData += ' ' + (initialPoint.x - size) + ',' + (initialPoint.y - 2 * size);
